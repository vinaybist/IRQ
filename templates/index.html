async function generateRequirements() { const securityControl = document.getElementById('security-control').value; const resultsSection = document.getElementById('results-section'); const requirementsContent = document.getElementById('requirements-content'); const loadingSpinner = document.getElementById('loading-spinner'); if (!securityControl.trim()) { alert('Please enter a security control'); return; } try { // Show results section and spinner, hide content resultsSection.classList.remove('hidden'); requirementsContent.classList.add('hidden'); loadingSpinner.classList.remove('hidden'); const response = await fetch('/analyze', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ security_control: securityControl }), }); const data = await response.json(); if (response.ok) { requirementsContent.innerHTML = marked.parse(data.requirements); } else { requirementsContent.innerHTML = `
Error: ${data.error}
`; } } catch (error) { console.error('Error:', error); requirementsContent.innerHTML = '
An error occurred while generating requirements
'; } finally { // Always hide spinner and show content after completion (success or error) loadingSpinner.classList.add('hidden'); requirementsContent.classList.remove('hidden'); } }
